// Helper function for course abbreviation detection
const getCourseAbbreviation = (courseName) => {
  if (courseName.includes('Information Technology')) return 'BSIT';
  if (courseName.includes('Computer Science')) return 'BSCS';
  if (courseName.includes('Information Systems')) return 'BSIS';
  if (courseName.includes('Computer Technology')) return 'ACT';
  return courseName.substring(0, 4).toUpperCase();
};

// Enhanced filtering with both program and year matching
const filteredSections = allSections.filter(section => {
  const programMatch = section.program === selectedProgram || 
                      section.program.includes(getCourseAbbreviation(selectedProgram));
  const yearMatch = section.year_level === yearNumber;
  return programMatch && yearMatch;
});
